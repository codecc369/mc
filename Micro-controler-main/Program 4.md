		AREA SQUARE, CODE, READONLY
	ENTRY
	START
		LDR R0,=Table ;
		LDR R2,=0X000000 ;
		LDR R1,=6 ;
		MOV R1,R1,LSL#0X2 ;
		ADD R0,R0,R1 ;
		LDR R3,[R0] ;
		NOP
		NOP
		NOP
	BACK B BACK
	TABLE	DCD 0X00000000;
			DCD 0X00000001;
   			DCD 0X00000004;
